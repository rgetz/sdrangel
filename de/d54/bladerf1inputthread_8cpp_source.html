69 void Bladerf1InputThread::run()
70 {
71  int res;
72 
73  m_running = true;
74  m_startWaiter.wakeAll();
75 
76  while(m_running) {
77  if((res = bladerf_sync_rx(m_dev, m_buf, BLADERF_BLOCKSIZE, NULL, 10000)) < 0) {
78  qCritical("BladerfThread: sync error: %s", strerror(errno));
79  break;
80  }
81 
82  callback(m_buf, 2 * BLADERF_BLOCKSIZE);
83  }
84 
85  m_running = false;
86 }
