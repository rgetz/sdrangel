122 int RTPPacketBuilder::AddCSRC(uint32_t csrc)
123 {
124  if (!init)
125  return ERR_RTP_PACKBUILD_NOTINIT;
126  if (numcsrcs >= RTP_MAXCSRCS)
127  return ERR_RTP_PACKBUILD_CSRCLISTFULL;
128 
129  int i;
130 
131  for (i = 0; i < numcsrcs; i++)
132  {
133  if (csrcs[i] == csrc)
134  return ERR_RTP_PACKBUILD_CSRCALREADYINLIST;
135  }
136  csrcs[numcsrcs] = csrc;
137  numcsrcs++;
138  return 0;
139 }

141 int RTPPacketBuilder::DeleteCSRC(uint32_t csrc)
142 {
143  if (!init)
144  return ERR_RTP_PACKBUILD_NOTINIT;
145 
146  int i = 0;
147  bool found = false;
148 
149  while (!found && i < numcsrcs)
150  {
151  if (csrcs[i] == csrc)
152  found = true;
153  else
154  i++;
155  }
156 
157  if (!found)
158  return ERR_RTP_PACKBUILD_CSRCNOTINLIST;
159 
160  // move the last csrc in the place of the deleted one
161  numcsrcs--;
162  if (numcsrcs > 0 && numcsrcs != i)
163  csrcs[i] = csrcs[numcsrcs];
164  return 0;
165 }

248 int RTPPacketBuilder::PrivateBuildPacket(const void *data, unsigned int len, uint8_t pt, bool mark, uint32_t timestampinc, bool gotextension, uint16_t hdrextID, const void *hdrextdata,
249  unsigned int numhdrextwords)
250 {
251  RTPPacket p(pt, data, len, seqnr, timestamp, ssrc, mark, numcsrcs, csrcs, gotextension, hdrextID, (uint16_t) numhdrextwords, hdrextdata, buffer, maxpacksize);
252  int status = p.GetCreationError();
253 
254  if (status < 0)
255  return status;
256  packetlength = p.GetPacketLength();
257 
258  if (numpackets == 0) // first packet
259  {
260  lastwallclocktime = RTPTime::CurrentTime();
261  lastrtptimestamp = timestamp;
262  prevrtptimestamp = timestamp;
263  }
264  else if (timestamp != prevrtptimestamp)
265  {
266  lastwallclocktime = RTPTime::CurrentTime();
267  lastrtptimestamp = timestamp;
268  prevrtptimestamp = timestamp;
269  }
270 
271  numpayloadbytes += (uint32_t) p.GetPayloadLength();
272  numpackets++;
273  timestamp += timestampinc;
274  seqnr++;
275 
276  //qDebug("RTPPacketBuilder::PrivateBuildPacket: numpackets: %u timestamp: %u timestampinc: %u seqnr: %u", numpackets, timestamp, timestampinc, seqnr);
277
