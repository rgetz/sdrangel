114 inline int RTPInternalSourceData::SetRTPDataAddress(const RTPAddress *a)
115 {
116  if (a == 0)
117  {
118  if (rtpaddr)
119  {
120  delete rtpaddr;
121  rtpaddr = 0;
122  }
123  }
124  else
125  {
126  RTPAddress *newaddr = a->CreateCopy();
127  if (newaddr == 0)
128  return ERR_RTP_OUTOFMEM;
129 
130  if (rtpaddr && a != rtpaddr)
131  delete rtpaddr;
132  rtpaddr = newaddr;
133  }
134  isrtpaddrset = true;
135  return 0;
136 }
