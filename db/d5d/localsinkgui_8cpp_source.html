83 bool LocalSinkGUI::handleMessage(const Message& message)
84 {
85  if (LocalSink::MsgSampleRateNotification::match(message))
86  {
87  LocalSink::MsgSampleRateNotification& notif = (LocalSink::MsgSampleRateNotification&) message;
88  //m_channelMarker.setBandwidth(notif.getSampleRate());
89  m_sampleRate = notif.getSampleRate();
90  displayRateAndShift();
91  return true;
92  }
93  else if (LocalSink::MsgConfigureLocalSink::match(message))
94  {
95  const LocalSink::MsgConfigureLocalSink& cfg = (LocalSink::MsgConfigureLocalSink&) message;
96  m_settings = cfg.getSettings();
97  blockApplySettings(true);
98  displaySettings();
99  blockApplySettings(false);
100  return true;
101  }
102  else
103  {
104  return false;
105  }
106 }

200 void LocalSinkGUI::displayRateAndShift()
201 {
202  int shift = m_shiftFrequencyFactor * m_sampleRate;
203  double channelSampleRate = ((double) m_sampleRate) / (1<<m_settings.m_log2Decim);
204  QLocale loc;
205  ui->offsetFrequencyText->setText(tr("%1 Hz").arg(loc.toString(shift)));
206  ui->channelRateText->setText(tr("%1k").arg(QString::number(channelSampleRate / 1000.0, 'g', 5)));
207  m_channelMarker.setCenterFrequency(shift);
208  m_channelMarker.setBandwidth(channelSampleRate);
209 }
