45 class RemoteOutputExpAvg {
46 public:
47  RemoteOutputExpAvg(float alpha) :
48  m_alpha(alpha),
49  m_start(true),
50  m_s(0)
51  {}
52  int put(int y)
53  {
54  if (m_start) {
55  m_start = false;
56  m_s = y;
57  } else {
58  m_s = m_alpha*y + (1.0-m_alpha)*m_s;
59  }
60  return roundf(m_s);
61  }
62  void reset() {
63  m_start = true;
64  }
65 
66 private:
67  float m_alpha;
68  bool m_start;
69  float m_s;
70 };
