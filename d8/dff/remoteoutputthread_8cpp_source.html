120 void RemoteOutputThread::tick()
121 {
122  if (m_running)
123  {
124  qint64 throttlems = m_elapsedTimer.restart();
125 
126  if (throttlems != m_throttlems)
127  {
128  m_throttlems = throttlems;
129  m_samplesChunkSize = (m_samplerate * (m_throttlems+(m_throttleToggle ? 1 : 0))) / 1000;
130  m_samplesChunkSize = m_samplesChunkSize + m_chunkCorrection > 0 ? m_samplesChunkSize + m_chunkCorrection : m_samplesChunkSize;
131  m_throttleToggle = !m_throttleToggle;
132  }
133 
134  SampleVector::iterator readUntil;
135 
136  m_sampleFifo->readAdvance(readUntil, m_samplesChunkSize); // pull samples
137  SampleVector::iterator beginRead = readUntil - m_samplesChunkSize;
138  m_samplesCount += m_samplesChunkSize;
139 
140  m_udpSinkFEC.write(beginRead, m_samplesChunkSize);
141  }
142 }
