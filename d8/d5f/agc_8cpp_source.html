108 double MagAGC::feedAndGetValue(const Complex& ci)
109 {
110  m_magsq = ci.real()*ci.real() + ci.imag()*ci.imag();
111  m_moving_average.feed(m_magsq);
112 
113  if (m_clamping)
114  {
115  if (m_squared)
116  {
117  if (m_magsq > m_clampMax) {
118  m_u0 = m_clampMax / m_magsq;
119  } else {
120  m_u0 = m_R / m_moving_average.average();
121  }
122  }
123  else
124  {
125  if (sqrt(m_magsq) > m_clampMax) {
126  m_u0 = m_clampMax / sqrt(m_magsq);
127  } else {
128  m_u0 = m_R / sqrt(m_moving_average.average());
129  }
130  }
131  }
132  else
133  {
134  m_u0 = m_R / (m_squared ? m_moving_average.average() : sqrt(m_moving_average.average()));
135  }
136 
137  if (m_thresholdEnable)
138  {
139  bool open = false;
140 
141  if (m_magsq > m_threshold)
142  {
143  if (m_gateCounter < m_gate) {
144  m_gateCounter++;
145  } else {

147  }
148  }
149  else

152  }
153 
154  if (open)

156  m_count = m_stepDownDelay; // delay before step down (grace delay)
157  }
158  else
159  {

162  }
163
