119 void NFMMod::pull(Sample& sample)
120 {
121  if (m_settings.m_channelMute)
122  {
123  sample.m_real = 0.0f;
124  sample.m_imag = 0.0f;
125  return;
126  }
127 
128  Complex ci;
129 
130  m_settingsMutex.lock();
131 
132  if (m_interpolatorDistance > 1.0f) // decimate
133  {
134  modulateSample();
135 
136  while (!m_interpolator.decimate(&m_interpolatorDistanceRemain, m_modSample, &ci))
137  {
138  modulateSample();
139  }
140  }
141  else
142  {
143  if (m_interpolator.interpolate(&m_interpolatorDistanceRemain, m_modSample, &ci))
144  {
145  modulateSample();
146  }
147  }
148 
149  m_interpolatorDistanceRemain += m_interpolatorDistance;
150 
151  ci *= m_carrierNco.nextIQ(); // shift to carrier frequency
152 
153  m_settingsMutex.unlock();
154 
155  double magsq = ci.real() * ci.real() + ci.imag() * ci.imag();
156  magsq /= (SDR_TX_SCALED*SDR_TX_SCALED);
157  m_movingAverage(magsq);
158  m_magsq = m_movingAverage.asDouble();
159 
160  sample.m_real = (FixReal) ci.real();
161  sample.m_imag = (FixReal) ci.imag();
162 }

177 void NFMMod::modulateSample()
178 {
179  Real t;
180 
181  pullAF(t);
182 
183  if (m_settings.m_feedbackAudioEnable) {
184  pushFeedback(t * m_settings.m_feedbackVolumeFactor * 16384.0f);
185  }
186 
187  calculateLevel(t);
188  m_audioBufferFill++;
189 
190  if (m_settings.m_ctcssOn)
191  {
192  m_modPhasor += (m_settings.m_fmDeviation / (float) m_audioSampleRate) * (0.85f * m_bandpass.filter(t) + 0.15f * 378.0f * m_ctcssNco.next()) * (M_PI / 378.0f);
193  }
194  else
195  {
196  // 378 = 302 * 1.25; 302 = number of filter taps (established experimentally)
197  m_modPhasor += (m_settings.m_fmDeviation / (float) m_audioSampleRate) * m_bandpass.filter(t) * (M_PI / 378.0f);
198  }
199 
200  m_modSample.real(cos(m_modPhasor) * 0.891235351562f * SDR_TX_SCALEF); // -1 dB
201  m_modSample.imag(sin(m_modPhasor) * 0.891235351562f * SDR_TX_SCALEF);
202 }

204 void NFMMod::pullAF(Real& sample)
205 {
206  switch (m_settings.m_modAFInput)
207  {
208  case NFMModSettings::NFMModInputTone:
209  sample = m_toneNco.next();
210  break;
211  case NFMModSettings::NFMModInputFile:
212  // sox f4exb_call.wav --encoding float --endian little f4exb_call.raw
213  // ffplay -f f32le -ar 48k -ac 1 f4exb_call.raw
214  if (m_ifstream.is_open())
215  {
216  if (m_ifstream.eof())
217  {
218  if (m_settings.m_playLoop)
219  {
220  m_ifstream.clear();
221  m_ifstream.seekg(0, std::ios::beg);
222  }
223  }
224 
225  if (m_ifstream.eof())
226  {
227  sample = 0.0f;
228  }
229  else
230  {
231  m_ifstream.read(reinterpret_cast<char*>(&sample), sizeof(Real));
232  sample *= m_settings.m_volumeFactor;
233  }
234  }
235  else
236  {
237  sample = 0.0f;
238  }
239  break;
240  case NFMModSettings::NFMModInputAudio:
241  sample = ((m_audioBuffer[m_audioBufferFill].l + m_audioBuffer[m_audioBufferFill].r) / 65536.0f) * m_settings.m_volumeFactor;
242  break;
243  case NFMModSettings::NFMModInputCWTone:
244  Real fadeFactor;
245 
246  if (m_cwKeyer.getSample())
247  {
248  m_cwKeyer.getCWSmoother().getFadeSample(true, fadeFactor);
249  sample = m_toneNco.next() * fadeFactor;
250  }
251  else
252  {
253  if (m_cwKeyer.getCWSmoother().getFadeSample(false, fadeFactor))
254  {
255  sample = m_toneNco.next() * fadeFactor;
256  }
257  else
258  {
259  sample = 0.0f;
260  m_toneNco.setPhase(0);
261  }
262  }
263  break;
264  case NFMModSettings::NFMModInputNone:
265  default:
266  sample = 0.0f;
267  break;
268  }

271 void NFMMod::pushFeedback(Real sample)
272 {
273  Complex c(sample, sample);
274  Complex ci;
275 
276  if (m_feedbackInterpolatorDistance < 1.0f) // interpolate
277  {
278  while (!m_feedbackInterpolator.interpolate(&m_feedbackInterpolatorDistanceRemain, c, &ci))
279  {
280  processOneSample(ci);
281  m_feedbackInterpolatorDistanceRemain += m_feedbackInterpolatorDistance;
282  }
283  }
284  else // decimate
285  {
286  if (m_feedbackInterpolator.decimate(&m_feedbackInterpolatorDistanceRemain, c, &ci))
287  {
288  processOneSample(ci);
289  m_feedbackInterpolatorDistanceRemain += m_feedbackInterpolatorDistance;
290  }
291  }

294 void NFMMod::processOneSample(Complex& ci)
295 {
296  m_feedbackAudioBuffer[m_feedbackAudioBufferFill].l = ci.real();
297  m_feedbackAudioBuffer[m_feedbackAudioBufferFill].r = ci.imag();
298  ++m_feedbackAudioBufferFill;
299 
300  if (m_feedbackAudioBufferFill >= m_feedbackAudioBuffer.size())
301  {
302  uint res = m_feedbackAudioFifo.write((const quint8*)&m_feedbackAudioBuffer[0], m_feedbackAudioBufferFill);
303 
304  if (res != m_feedbackAudioBufferFill)
305  {
306  qDebug("AMDemod::pushFeedback: %u/%u audio samples written m_feedbackInterpolatorDistance: %f",
307  res, m_feedbackAudioBufferFill, m_feedbackInterpolatorDistance);
308  m_feedbackAudioFifo.clear();
309  }
310 
311  m_feedbackAudioBufferFill = 0;
312  }

315 void NFMMod::calculateLevel(Real& sample)
316 {
317  if (m_levelCalcCount < m_levelNbSamples)
318  {
319  m_peakLevel = std::max(std::fabs(m_peakLevel), sample);
320  m_levelSum += sample * sample;
321  m_levelCalcCount++;
322  }
323  else
324  {
325  qreal rmsLevel = sqrt(m_levelSum / m_levelNbSamples);
326  //qDebug("NFMMod::calculateLevel: %f %f", rmsLevel, m_peakLevel);
327  emit levelChanged(rmsLevel, m_peakLevel, m_levelNbSamples);
328  m_peakLevel = 0.0f;
329  m_levelSum = 0.0f;
330  m_levelCalcCount = 0;
331  }

446  }
447 
448  return true;
449  }
450  else if (DSPSignalNotification::match(cmd))
451  {
452  return true;
453  }
454  else
455  {
456  return false;
457  }
