97 template <typename Type, uint32_t Order>
98 Type IIRFilter<Type, Order>::run(const Type& sample)
99 {
100  Type y = m_b[0] * sample;
101 
102  for (uint32_t i = Order; i > 0; i--)
103  {
104  y += m_b[i] * m_x[i-1] + m_a[i] * m_y[i-1];
105 
106  if (i > 1) // shift
107  {
108  m_x[i-1] = m_x[i-2];
109  m_y[i-1] = m_y[i-2];
110  }
111  }
112 
113  // last shift
114  m_x[0] = sample;
115  m_y[0] = y;
116 
117  return y;
118 }
