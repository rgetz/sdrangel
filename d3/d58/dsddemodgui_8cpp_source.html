526 void DSDDemodGUI::tick()
527 {
528  double magsqAvg, magsqPeak;
529  int nbMagsqSamples;
530  m_dsdDemod->getMagSqLevels(magsqAvg, magsqPeak, nbMagsqSamples);
531  double powDbAvg = CalcDb::dbPower(magsqAvg);
532  double powDbPeak = CalcDb::dbPower(magsqPeak);
533 
534  ui->channelPowerMeter->levelChanged(
535  (100.0f + powDbAvg) / 100.0f,
536  (100.0f + powDbPeak) / 100.0f,
537  nbMagsqSamples);
538 
539  if (m_tickCount % 4 == 0) {
540  ui->channelPower->setText(tr("%1 dB").arg(powDbAvg, 0, 'f', 1));
541  }
542 
543  bool squelchOpen = m_dsdDemod->getSquelchOpen();
544 
545  if (squelchOpen != m_squelchOpen)
546  {
547  if (squelchOpen) {
548  ui->audioMute->setStyleSheet("QToolButton { background-color : green; }");
549  } else {
550  ui->audioMute->setStyleSheet("QToolButton { background:rgb(79,79,79); }");
551  }
552 
553  m_squelchOpen = squelchOpen;
554  }
555 
556  // "slow" updates
557 
558  if (m_tickCount % 10 == 0)
559  {
560  ui->inLevelText->setText(QString::number(m_dsdDemod->getDecoder().getInLevel()));
561  ui->inCarrierPosText->setText(QString::number(m_dsdDemod->getDecoder().getCarrierPos()));
562  ui->zcPosText->setText(QString::number(m_dsdDemod->getDecoder().getZeroCrossingPos()));
563  ui->symbolSyncQualityText->setText(QString::number(m_dsdDemod->getDecoder().getSymbolSyncQuality()));
564 
565  if (m_dsdDemod->getDecoder().getVoice1On()) {
566  ui->slot1On->setStyleSheet("QToolButton { background-color : green; }");
567  } else {
568  ui->slot1On->setStyleSheet("QToolButton { background-color : rgb(79,79,79); }");
569  }
570 
571  if (m_dsdDemod->getDecoder().getVoice2On()) {
572  ui->slot2On->setStyleSheet("QToolButton { background-color : green; }");
573  } else {
574  ui->slot2On->setStyleSheet("QToolButton { background-color : rgb(79,79,79); }");
575  }
576 
577  const char *frameTypeText = m_dsdDemod->getDecoder().getFrameTypeText();
578 
579  if (frameTypeText[0] == '\0') {
580  ui->syncText->setStyleSheet("QLabel { background:rgb(53,53,53); }"); // turn off background
581  } else {
582  ui->syncText->setStyleSheet("QLabel { background:rgb(37,53,39); }"); // turn on background
583  }
584 
585  ui->syncText->setText(QString(frameTypeText));
586 
587  const char *formatStatusText = m_dsdDemod->updateAndGetStatusText();
588  ui->formatStatusText->setText(QString(formatStatusText));
589 
590  if (ui->activateStatusLog->isChecked()) {
591  m_dsdStatusTextDialog.addLine(QString(formatStatusText));
592  }
593 
594  if (formatStatusText[0] == '\0') {
595  ui->formatStatusText->setStyleSheet("QLabel { background:rgb(53,53,53); }"); // turn off background
596  } else {
597  ui->formatStatusText->setStyleSheet("QLabel { background:rgb(37,53,39); }"); // turn on background
598  }
