68  inline cmplx frrc(float fb, float a, int i, int len)
69  {
70  float x = i/(float)len; // normalize to [0..1]
71  x = 0.5-fabs(x-0.5); // apply symmetry: now both halves overlap near 0
72  float tr = fb*a; // half the transition zone
73 
74  if (x < fb-tr)

82  }
83  else
