775 inline RTPPacket *RTPSourceData::GetNextPacket()
776 {
777  if (!validated)
778  return 0;
779 
780  RTPPacket *p;
781 
782  if (packetlist.empty())
783  return 0;
784  p = *(packetlist.begin());
785  packetlist.pop_front();
786  return p;
787 }

789 inline void RTPSourceData::FlushPackets()
790 {
791  std::list<RTPPacket *>::const_iterator it;
792 
793  for (it = packetlist.begin(); it != packetlist.end(); ++it)
