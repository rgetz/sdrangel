298 void AudioDialogX::updateOutputSDPString()
299 {
300  QString format;
301  int nChannels = m_outputDeviceInfo.udpChannelMode == AudioOutput::UDPChannelStereo ? 2 : 1;
302  uint32_t effectiveSampleRate = m_outputDeviceInfo.sampleRate / (m_outputDeviceInfo.udpDecimationFactor == 0 ? 1 : m_outputDeviceInfo.udpDecimationFactor);
303 
304  switch (m_outputDeviceInfo.udpChannelCodec)
305  {
306  case AudioOutput::UDPCodecALaw:
307  format = "PCMA";
308  break;
309  case AudioOutput::UDPCodecULaw:
310  format = "PCMU";
311  break;
312  case AudioOutput::UDPCodecG722:
313  format = "G722";
314  effectiveSampleRate /= 2; // codec does a decimation by 2
315  break;
316  case AudioOutput::UDPCodecL8:
317  format = "L8";
318  break;
319  case AudioOutput::UDPCodecOpus:
320  format = "opus";
321  nChannels = 2; // always 2 even for mono
322  effectiveSampleRate = 48000; // always 48000 regardless of input rate
323  break;
324  case AudioOutput::UDPCodecL16:
325  default:
326  format = "L16";
327  break;
328  }
329 
330  ui->outputSDPText->setText(tr("%1/%2/%3").arg(format).arg(effectiveSampleRate).arg(nChannels));
331 }

333 void AudioDialogX::check()
334 {
335  int nChannels = m_outputDeviceInfo.udpChannelMode == AudioOutput::UDPChannelStereo ? 2 : 1;
336  uint32_t decimationFactor = m_outputDeviceInfo.udpDecimationFactor == 0 ? 1 : m_outputDeviceInfo.udpDecimationFactor;
337 
338  if (m_outputDeviceInfo.udpChannelCodec == AudioOutput::UDPCodecALaw)
339  {
340  if ((nChannels != 1) || (m_outputDeviceInfo.sampleRate/decimationFactor != 8000)) {
341  QMessageBox::information(this, tr("Message"), tr("PCMA must be 8000 Hz single channel"));
342  }
343  }
344  else if (m_outputDeviceInfo.udpChannelCodec == AudioOutput::UDPCodecULaw)
345  {
346  if ((nChannels != 1) || (m_outputDeviceInfo.sampleRate/decimationFactor != 8000)) {
347  QMessageBox::information(this, tr("Message"), tr("PCMU must be 8000 Hz single channel"));
348  }
349  }
350  else if (m_outputDeviceInfo.udpChannelCodec == AudioOutput::UDPCodecG722)
351  {
352  if ((nChannels != 1) || (m_outputDeviceInfo.sampleRate/decimationFactor != 16000)) {
353  QMessageBox::information(this, tr("Message"), tr("G722 must be 16000 Hz single channel"));
354  }
355  }
356  else if (m_outputDeviceInfo.udpChannelCodec == AudioOutput::UDPCodecOpus)
357  {
358  int effectiveSampleRate = m_outputDeviceInfo.sampleRate/decimationFactor;
359  if ((effectiveSampleRate != 48000) && (effectiveSampleRate != 24000) && (effectiveSampleRate != 16000) && (effectiveSampleRate != 12000)) {
360  QMessageBox::information(this, tr("Message"), tr("Opus takes only 48, 24, 16 or 12 kHz sample rates"));
361  }
362  }
363 }
