220  int SetString(uint8_t **dest, std::size_t *destlen, const uint8_t *s, std::size_t len)
221  {
222  if (len <= 0)
223  {
224  if (*dest)
225  delete[] (*dest);
226  *dest = 0;
227  *destlen = 0;
228  }
229  else
230  {
231  len = (len > RTCP_SDES_MAXITEMLENGTH) ? RTCP_SDES_MAXITEMLENGTH : len;
232  uint8_t *str2 = new uint8_t[len];
233  memcpy(str2, s, len);
234  *destlen = len;
235  if (*dest)
236  delete[] (*dest);
237  *dest = str2;
238  }
239  return 0;
240  }
