44 // Process samples.
45 void LowPassFilterRC::process(const Real& sample_in, Real& sample_out)
46 {

48  * Continuous domain:
49  * H(s) = 1 / (1 - s * timeconst)
50  *
51  * Discrete domain:
52  * H(z) = (1 - exp(-1/timeconst)) / (1 - exp(-1/timeconst) / z)
53  */
54 
55  m_y1 = (sample_in * m_b0) - (m_y1 * m_a1);
